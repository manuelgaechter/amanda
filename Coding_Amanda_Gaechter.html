Übung 4: Mit For-Loop Monte-Carlo-Simulation durchführen

In [26]:
def point_offset(point, distance = 100):
    x_alt = point.x
    y_alt = point.y
    
    distance = int(distance)
    
    x_neu = x_alt + random.randrange(-distance,distance)
    y_neu = y_alt + random.randrange(-distance,distance)
    
    point_off = Point(x_neu, y_neu)
    
    return(point_off)

In [35]:
   montecarlo =[]

for runs in range(50): #0

 

    zeckenstiche_sim = gpd.GeoDataFrame() #1

    zeckenstiche_sim["geometry"] = zeckenstiche["geometry"].apply(point_offset) #2

    zeckenstiche_sim["ID"]=zeckenstiche["ID"] #3

    zeckenstiche_sim["Run_Nr"]=runs #4

    montecarlo.append(zeckenstiche_sim) #5
#0 definiert 50 runs für die nachfolgende Aktion #1 erstellt ein leeres GeoDataFrame #2 erstellt Spalte und generiert mit Funktion "point offset" veränderte Koordinaten #3 kopiert Spalte "ID" aus Originaldaten in das neue DataFrame #4 erstellt Spalte "Run_Nr", die auf Funktion "runs" mit 50 Durchgängen verweist # "montecarlo" erstellt Liste der 50 Durchgänge "montecarlo" enthält nun 50 dataframes mit random verschobenen Punkten
